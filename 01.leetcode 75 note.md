### 题目解析

1st
```cpp=
#include<bits/stdc++.h>
using namespace std;
int main(){
	double x1,y1,x2,y2,x3,y3,x4,y4,x,y;
	while(cin>>x1>>y1>>x2>>y2>>x3>>y3>>x4>>y4){
		//(1,3)
		if(x1==x3 && y1==y3){
			x=(x2+x4)-x1;
			y=(y2+y4)-y1;
		}//(1,4)
		else if(x1==x4 && y1==y4){
			x=(x2+x3)-x1;
			y=(y2+y3)-y1;
		}//(2,3)
		else if(x2==x3 && y2==y3){
			x=(x1+x4)-x2;
			y=(y1+y4)-y2;
		}//(2,4)
		else{
			x=(x1+x3)-x2;
			y=(y1+y3)-y2;
		}
		//output
		cout<<fixed<<setprecision(3)<<x<<" "<<y<<endl;
	}
	
	return 0;
}
```
tips- if判断跟着(1,3)(1,4)(2,3)(2,4) 计算用 先x2,x4(1,3)两数相加后减x1/y1或x2/y2
